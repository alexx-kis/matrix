# rewriting promise code with async/await

- нужно понимать, как код на `.then()/.catch()` переписывается на `async/await` и наоборот

- важно помнить:

```js
promise
  .then(...)
  .catch(...)

// и

try {
  await promise;
} catch (error) {
  //
}
```

- решают одну и ту же задачу

## самый простой пример

```js
function getUserName() {
  return Promise.resolve('Alex');
}

// promise

getUserName().then((userName) => {
  console.log(userName);
});

// async/await

async function printUserName() {
  const userName = await getUserName;

  console.log(userName);
}

printUserName();
```

## цепочка из нескольких then

```js
function getNumberValue() {
  return Promise.resolve(10);
}

// promise

getNumberValue()
  .then((numberValue) => {
    return numberValue * 2;
  })
  .then((resultValue) => {
    console.log(resultValue);
  });

// async/await

async function calculateValue() {
  const numberValue = await getNumberValue();

  const resultValue = numberValue * 2;

  console.log(resultValue);
}

calculateValue();
```

## несколько последовательных запросов

```js
function getUserData() {
  return Promise.resolve({
    id: 5,
  });
}

function getOrderData(userId) {
  return Promise.resolve([`Order for user ${userId}`]);
}

// promise

getUserData()
  .then((userData) => {
    return getOrderData(userData.id);
  })
  .then((orderData) => {
    console.log(orderData);
  });

// async/await

async function loadOrderData() {
  const userData = await getUserData();

  const orderData = await getOrderData(userData.id);

  console.log(orderData);
}

loadOrderData();
```

## переписывание catch

```js
// promise

getUserData()
  .then((userData) => {
    console.log(userData);
  })
  .catch((requestError) => {
    console.log(requestError);
  });

// async/await

async function loadUserData() {
  try {
    const userData = await getUserData();

    console.log(userData);
  } catch (requestError) {
    console.log(requestError);
  }
}
```

## переписывание finally

```js
// promise

getUserData()
  .then((userData) => {
    console.log(userData);
  })
  .catch((requestError) => {
    console.log(requestError);
  })
  .finally(() => {
    console.log('finished');
  });

// async/await

async function loadUserData() {
  try {
    const userData = await getUserData();
    console.log(userData);
  } catch (requestError) {
    console.log(requestError);
  } finally {
    console.log('finished');
  }
}
```

## частая ошибка

```js
getProductData()
  .then((productData) => {
    return productData.price;
  })
  .then((priceValue) => {
    console.log(priceValue);
  });

// неправильно:

async function showPriceValue() {
  await getProductData().price;
}

// правильно:

async function showPriceValue() {
  const productData = await getProductData();

  console.log(productData.price);
}
```

- сначала нужно дождаться promise, потом работать с результатом

## `Promise.all` -> `async/await`

```js
// promise

Promise.all([loadProfileData(), loadSettingsData()]).then((resultsArray) => {
  console.log(resultsArray);
});

// async/await

async function loadPageData() {
  const resultsArray = await Promise.all([loadProfileData(), loadSettingsData()]);
  console.log(resultsArray);
}
```

- `Promise.all` никуда не исчезает, `async/await` не заменяют его

## ошибка при механическом переписывании

```js
// было:

getUserData()
  .then((userData) => {
    return getOrdersData(userData.id);
  })
  .then((ordersData) => {
    console.log(ordersData);
  });

// некоторые пишут:

async function loadData() {
  const userData = await getUserData();

  getOrdersData(userData.id);

  console.log('Done');
}

// ошибка в том, что второй promise никто не ждет

// правильно:

async function loadData() {
  const userData = await getUserData();

  const ordersData = await getOrdersData(userData.id);

  console.log(ordersData);
}
```

## когда async/await не делает код лучше

- иногда promise-цепочка короче

```js
fetch('/api/users').then((responseObject) => responseObject.json());
```

## как обычно спрашивают на собеседовании

```js
fetch('/api/profile')
  .then((responseObject) => responseObject.json())
  .then((profileData) => {
    console.log(profileData);
  })
  .catch((requestError) => {
    console.log(requestError);
  });

// нужно переписать на async/await
// ответ:

async function loadData() {
  try {
    const responseObject = await fetch('/api/profile');
    const profileData = await responseObject.json();
    console.log(profileData);
  } catch (requestError) {
    console.log(requestError);
  }
}

loadData();
```
